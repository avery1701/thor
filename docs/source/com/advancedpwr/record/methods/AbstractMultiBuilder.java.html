16 package com.advancedpwr.record.methods;
17 
18 import java.util.Iterator;
19 import java.util.List;
20 
21 import com.advancedpwr.record.AccessPath;
22 
23 public abstract class AbstractMultiBuilder extends BuildMethodWriter
24 {
25     protected MethodBuilderFactory fieldCacheFactory;
26     
27     public MethodBuilderFactory getCacheFactory()
28     {
29         if ( fieldCacheFactory == null )
30         {
31             fieldCacheFactory = new MethodBuilderFactory();
32             
33         }
34         fieldCacheFactory.getBuilderCache().putAll( getFactory().getBuilderCache() );
35         return fieldCacheFactory;
36     }
37     
38     protected BuildMethodWriter createCachedInstanceMethodBuilder( AccessPath result )
39     {
40         BuildMethodWriter builder = getFactory().createMethodBuilder( result );
41         if ( getCacheFactory().contains( result ) )
42         {
43             builder = getCacheFactory().createMethodBuilder( result );
44         }
45         builder.setClassWriter( getClassWriter() );
46         cache( builder );
47         return builder;
48     }
49 
50     protected void cache( BuildMethodWriter inBuilder )
51     {
52         if ( getCacheFactory().contains( inBuilder.getAccessPath() ) )
53         {
54             return;
55         }
56         getCacheFactory().storeBuilder( inBuilder );
57         List<AccessPath> paths = inBuilder.getInstanceTree().getAccessPaths();
58         for ( Iterator iterator = paths.iterator(); iterator.hasNext(); )
59         {
60             AccessPath accessPath = (AccessPath) iterator.next();
61             BuildMethodWriter builder = getCacheFactory().createMethodBuilder( accessPath );
62             cache( builder );
63         }
64     }
65 }
