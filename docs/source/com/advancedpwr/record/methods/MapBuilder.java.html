16 package com.advancedpwr.record.methods;
17 
18 import com.advancedpwr.record.AccessPath;
19 import com.advancedpwr.record.inspect.MapPutPath;
20 
21 public class MapBuilder extends AbstractMultiBuilder
22 {
23 
24     protected void writePopulators()
25     {
26         for ( AccessPath result : getInstanceTree().getAccessPaths() )
27         {
28             if ( result instanceof MapPutPath )
29             {
30                 MapPutPath path = (MapPutPath)result;
31                 BuildMethodWriter keyBuilder = createKeyBuilder( path );
32                 BuildMethodWriter builder = createCachedInstanceMethodBuilder( result );
33                 writeLine( instanceName() + ".put( " + keyBuilder.resultBuilder() + ", " + builder.resultBuilder() + " )" );
34             }
35         }
36     }
37 
38     protected BuildMethodWriter createKeyBuilder( MapPutPath path )
39     {
40         AccessPath accessPath = new AccessPath();
41         accessPath.setTree( path.getKeyTree() );
42         return createCachedInstanceMethodBuilder( accessPath );
43 //      return getFactory().createMethodBuilder( accessPath );
44     }
45 }
