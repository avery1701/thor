36 /**
37  * An {@link ObjectRecorder} that records the <i>state</i> of an object tree as a Java class file. This recorder uses 
38  * reflection to inspect an object tree for Java Bean style setter / getter accessor methods, therefore it is only 
39  * capable of reconstructing an object tree through that convention.  Many web service return data structures adhere
40  * to the Java Bean accessor convention, so this class may be of use in recording a web service response for use in 
41  * unit testing.
42  * <p>
43  * Recording example:
44  * </p>
45  *<p><blockquote><pre>
46     Person person = new Person();
47     person.setName( "Jim" );
48     Person dad = new Person();
49     dad.setName( "John" );
50     person.setDad( dad );
51     
52     BeanRecorder recorder = new BeanRecorder();
53     recorder.setDestination( "recordings" );
54     recorder.record( person );
55  * </pre></blockquote><p>
56  * 
57  * The above example will record the object tree of the "person" instance as a Java class:
58  * 
59  * <p><blockquote><pre>
60     public class PersonFactory
61     {
62     
63         protected Person person;
64     
65         public Person buildPerson()
66         {
67             person = new Person();
68             person.setDad( buildPerson_1_1() );
69             person.setName( "Jim" );
70             return person;
71         }
72     
73         protected Person person_1_1;
74     
75         protected Person buildPerson_1_1()
76         {
77             dad_1_1 = new Person();
78             dad_1_1.setName( "John" );
79             return person_1_1;
80         }
81     
82     }
83  * </pre></blockquote><p>
84  * 
85  * To reconstruct the instance of "person" in a unit test:
86  * 
87  * <p><blockquote><pre>
88     Person person = new PersonFactory().buildPerson();
89  * </pre></blockquote><p>
90  * 
91  * The BeanRecorder is "instance aware" and supports {@link Collection} and {@link Map} objects.
92  * 
93  * @author Matthew Avery, mavery@advancedpwr.com on Jun 22, 2010
94  *
95  */
96 public class BeanRecorder extends AbstractRecorder
97 {
98     protected InstanceTree fieldInstanceTree;
99     protected MethodBuilderFactory fieldFactoryBuilder;
100    protected Set<Class> fieldStopClasses;
101    
102    public BeanRecorder()
103    {
104        setSuperClass( BaseFactory.class );
105    }
106    
107    protected void setObject( Object object )
108    {
109        if ( object == null )
110        {
111            throw new RecorderException( "Called setObject with null argument");
112        }
113        fieldInstanceTree = createInstanceTree( object );
114    }
115
116    protected InstanceTree createInstanceTree( Object object )
117    {
118        return new InstanceTree( getStopClasses(), object );
119    }
120
121    /* (non-Javadoc)
122     * @see com.advancedpwr.record.ObjectRecorder#record(java.lang.Object)
123     */
124    public <T> T record( T inObject )
125    {
126        setObject( inObject );
127        writeObject();
128        closeFile();
129        return inObject;
130    }
131
132    protected void closeFile()
133    {
134        if ( getDestination() != null )
135        {
136            close( getJavaFileWriter() );
137        }
138    }
139    
140
141    protected void writeObjectBuilderMethod()
142    {
143        AccessPath result = new AccessPath();
144        result.setTree( getInstanceTree() );
145        BuildMethodWriter builder = createMethodBuilder( result );
146        builder.setScopePublic();
147        builder.buildMethod();
148    }
149
150    protected BuildMethodWriter createMethodBuilder( AccessPath result )
151    {
152        BuildMethodWriter builder = getFactoryBuilder().createMethodBuilder( result );
153        builder.setClassWriter( this );
154        return builder;
155    }
156    
157    protected Object getObject()
158    {
159        return getInstanceTree().getObject();
160    }
161
162    protected InstanceTree getInstanceTree()
163    {
164        return fieldInstanceTree;
165    }
166
167    protected Set<Class> classes()
168    {
169        Set<Class> classes = new LinkedHashSet<Class>();
170        if ( getSuperClass() != null )
171        {
172            classes.add( getSuperClass() );
173        }
174        classes.addAll( getInstanceTree().getFactory().classes() );
175        if ( classes.contains( java.util.Date.class ) && classes.contains( java.sql.Date.class ) )
176        {
177            classes.remove( java.sql.Date.class );
178        }
179        return classes;
180    }
181    
182    protected MethodBuilderFactory getFactoryBuilder()
183    {
184        if ( fieldFactoryBuilder == null )
185        {
186            fieldFactoryBuilder = createMethodBuilderFactory();
187            
188        }
189        return fieldFactoryBuilder;
190    }
191
192    protected MethodBuilderFactory createMethodBuilderFactory()
193    {
194        MethodBuilderFactory factory = new MethodBuilderFactory();
195        factory.addBuilderFactory( new CollectionBuilderFactory() );
196        factory.addBuilderFactory( new MapBuilderFactory() );
197        return factory;
198    }
199    
200    public void addBuilderFactory( MethodWriterFactory inFactory )
201    {
202        getFactoryBuilder().addBuilderFactory( inFactory );
203    }
204    
205    public PrintWriter getPrintWriter()
206    {
207        if ( fieldPrintWriter == null && getDestination() != null )
208        {
209            setWriter( getJavaFileWriter() );
210        }
211        return fieldPrintWriter;
212    }
213    
214    public void stopDescent( Class inClass )
215    {
216        getStopClasses().add( inClass );
217    }
218    
219    public Set getStopClasses()
220    {
221        if ( fieldStopClasses == null )
222        {
223            fieldStopClasses = new HashSet();
224            fieldStopClasses.add( Date.class );
225            fieldStopClasses.add( java.sql.Date.class );
226            fieldStopClasses.add( Calendar.class );
227            fieldStopClasses.add( GregorianCalendar.class );
228        }
229        return fieldStopClasses;
230    }
231    
232}
