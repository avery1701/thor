16 package com.advancedpwr.record;
17 
18 import java.io.File;
19 import java.io.FileWriter;
20 import java.io.IOException;
21 import java.io.Writer;
22 
23 /**
24  * <code>AbstractRecorder</code> is the common base class for recorders.  This class provides
25  * some convenience methods for specifying the output directory for recorded classes.  It also
26  * provides the default {@link ClassDescriptor} for the superclass {@link ClassWriter}.
27  * 
28  * It is not necessary for future class recorders to extend AbstractRecorder.  This abstract class
29  * exists for convenience only.
30  * 
31  * @author Matthew Avery, mavery@advancedpwr.com on Sep 30, 2010
32  */
33 public abstract class AbstractRecorder extends ClassWriter implements ObjectRecorder
34 {
35 
36     protected File fieldDestinationDirectory;
37     protected FileWriter fieldJavaFileWriter;
38 
39     protected FileWriter createFileWriter( final File inFile )
40     {
41         WrappedIO wrapper = new WrappedIO()
42         {
43             
44             public Object exec() throws IOException
45             {
46                 return new FileWriter( inFile );
47             }
48         };
49         return (FileWriter) wrapper.wrap();
50     }
51 
52     protected FileWriter getJavaFileWriter()
53     {
54         if ( fieldJavaFileWriter == null )
55         {
56             fieldJavaFileWriter = createFileWriter( javaFile() );
57         }
58     
59         return fieldJavaFileWriter;
60     }
61 
62     protected File parentDirectory()
63     {
64         File parent = new File( getDestination(), packagePath() );
65         parent.mkdirs();
66         return parent;
67     }
68 
69     protected File javaFile()
70     {
71         return new File( parentDirectory(), getClassName() + ".java");
72     }
73 
74     protected String packagePath()
75     {
76         return getPackageName().replaceAll( "[.]", "/" );
77     }
78 
79     public File getDestination()
80     {
81         return fieldDestinationDirectory;
82     }
83 
84     /**
85      * Set the generated output directory as a {@link File}
86      * 
87      * @param directory
88      */
89     public void setDestination( File testDirectory )
90     {
91         fieldDestinationDirectory = testDirectory;
92     }
93     
94     /**
95      * Set the generated output directory as the name of a {@link File}.
96      * 
97      * @param directory
98      */
99     public void setDestination( String inFileName )
100    {
101        setDestination( new File( inFileName ) );
102    }
103
104    protected ClassDescriptor createDefaultDescriptor()
105    {
106        if ( objectClass().isArray() )
107        {
108            return new ArrayClassDescriptor( objectClass() );
109        }
110        return new DefaultClassDescriptor( objectClass() );
111    }
112
113    protected Class objectClass()
114    {
115        return getObject().getClass();
116    }
117
118    protected void close( final Writer inWriter )
119    {
120        new WrappedIO()
121        {
122            public Object exec() throws IOException
123            {
124                inWriter.flush();
125                inWriter.close();
126                return null;
127            }
128        }.wrap();
129    }
130
131}
