16 package com.advancedpwr.record;
17 
18 public class DefaultClassDescriptor implements ClassDescriptor
19 {
20 
21     public static final String GENERATED = ".generated";
22 
23     public static final String FACTORY = "Factory";
24     
25     protected Class fieldClass;
26     
27     public DefaultClassDescriptor()
28     {
29         
30     }
31     
32     public DefaultClassDescriptor( Class inClass )
33     {
34         setClass( inClass );
35     }
36     
37     public Class subject()
38     {
39         return fieldClass;
40     }
41 
42     public void setClass( Class inClass )
43     {
44         fieldClass = inClass;
45     }
46 
47     public String getClassName()
48     {
49         return subject().getSimpleName() + FACTORY;
50     }
51 
52     public String getPackageName()
53     {
54         return subject().getPackage().getName() + GENERATED;
55     }
56     
57     public String toString()
58     {
59         return getPackageName() + "." + getClassName();
60     }
61 
62 }
