16 package com.advancedpwr.record;
17 
18 /**
19  * A {@link ClassDescriptor} for array factories.  If the object to be recorded is a Java array, this
20  * class descriptor will create an appropriately named class.
21  * 
22  * For example, if the target object to be recorded is an array of <code>com.example.Person</code> objects, this 
23  * {@link ClassDescriptor} will generate a class name of "PersonArrayFactory" in the package "com.example.generated".
24  * 
25  * @author Matthew Avery, mavery@advancedpwr.com on Sep 30, 2010
26  *
27  */
28 public class ArrayClassDescriptor extends DefaultClassDescriptor
29 {
30 
31 
32     public ArrayClassDescriptor( Class inClass )
33     {
34         super( inClass );
35     }
36 
37 
38     public static final String ARRAY = "Array";
39 
40 
41     public String getClassName()
42     {
43         return subject().getSimpleName() + ARRAY + FACTORY;
44     }
45 
46 
47     public Class subject()
48     {
49         return fieldClass.getComponentType();
50     }
51 
52 }
