15 public class BeanInspector extends Inspector
16 {
17     
18     protected List<Method> sortedMethods()
19     {
20         if ( getInstanceTree().isStop() )
21         {
22             return Collections.EMPTY_LIST;
23         }
24         Method[] methods = objectClass().getMethods();
25         List list = Arrays.asList( methods );
26         Collections.sort( list, new MethodNameComparator() );
27         Collections.sort( list, new CollectionMethodComparator() );
28         Collections.sort( list, new MapMethodComparator() );
29         Collections.sort( list, new ArrayMethodComparator() );
30         return list;
31     }
32     
33 
34     public void inspect( InstanceTree inTree )
35     {
36         setInstanceTree( inTree );
37         List<Method> methods = sortedMethods();
38         for ( Method method : methods )
39         {
40             setCurrentMethod( method );
41             addMethodAccessPath();
42         }
43         
44     }
45     
46     protected void addMethodAccessPath()
47     {
48         if ( isSetter() && hasGetterMethod() )
49         {
50             Method getter = getterMethod();
51             Object result = invoke( getter );
52             if ( result != null )
53             {
54                 addAccessPathForResult( getter, result );
55             }
56         }
57     }
58     
59     protected void addAccessPathForResult( Method inGetter, Object result )
60     {
61         AccessPath path = createAccessorMethodPath( inGetter, result );
62         addAccessPath( path );
63     }
64     
65     protected AccessPath createAccessorMethodPath( Method inGetter, Object result )
66     {
67         AccessorMethodPath accessor = new AccessorMethodPath();
68         accessor.setSetter( getCurrentMethod() );
69         accessor.setGetter( inGetter );
70         InstanceTree tree = createInstanceTree( result );
71         accessor.setTree( tree );
72         debug( "created accessor " + accessor + " for result " + result );
73         return accessor;
74     }
75 
76     protected boolean isSetter()
77     {
78         Method method = getCurrentMethod();
79         return Modifier.isPublic( method.getModifiers() )  && method.getName().startsWith( "set" ) && method.getParameterTypes().length == 1;
80     }
81     
82     protected String getterName()
83     {
84         if( boolean.class.equals( getCurrentMethod().getParameterTypes()[0] ) )
85         {
86             return getCurrentMethod().getName().replaceFirst( "set", "is" );
87         }
88         return getCurrentMethod().getName().replaceFirst( "set", "get" );
89     }
90     
91     protected Method getterMethod()
92     {
93         Method[] methods = objectClass().getMethods();
94         for ( int i = 0; i < methods.length; i++ )
95         {
96             Method method = methods[i];
97             if ( isGetter( method ) )
98             {
99                 return method;
100            }
101        }
102        return null;
103    }
104
105    protected boolean isGetter( Method method )
106    {
107        return method.getName().equals( getterName() ) && method.getParameterTypes().length == 0 && !Modifier.isStatic( method.getModifiers() );
108    }
109    
110    protected boolean hasGetterMethod()
111    {
112        return getterMethod() != null;
113    }
114
115    protected Method getCurrentMethod()
116    {
117        return getInstanceTree().getCurrentMethod();
118    }
119
120    protected void setCurrentMethod( Method currentMethod )
121    {
122        getInstanceTree().setCurrentMethod( currentMethod );
123    }
124    
125    protected Object invoke( Method getter )
126    {
127        try
128        {
129             return getter.invoke( getObject() );
130        }
131        catch ( Exception e )
132        {
133            throw new RecorderException( e );
134        }
135        
136    }
