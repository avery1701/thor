16 package com.advancedpwr.record.inspect;
17 
18 import java.lang.reflect.Method;
19 import java.util.Arrays;
20 import java.util.LinkedHashSet;
21 import java.util.Set;
22 
23 import com.advancedpwr.record.AccessPath;
24 
25 /**
26  * An {@link AccessPath} representing object access through Java Bean convention setter / getter accessor methods.
27  * 
28  * @author Matthew Avery, mavery@advancedpwr.com on May 29, 2010
29  *
30  */
31 public class AccessorMethodPath extends AccessPath
32 {
33 
34     protected Method fieldSetter;
35     
36     protected Method fieldGetter;
37 
38     public Method getSetter()
39     {
40         return fieldSetter;
41     }
42 
43     public void setSetter( Method setter )
44     {
45         fieldSetter = setter;
46     }
47 
48     
49     public String pathName()
50     {
51         return getSetter().getName();
52     }
53     
54     public Class getParameterClass()
55     {
56         return getSetter().getParameterTypes()[0];
57     }
58 
59     public Set<Class> getExceptions()
60     {
61         LinkedHashSet<Class> exceptions = new LinkedHashSet<Class>();
62         Class[] types = getSetter().getExceptionTypes();
63         exceptions.addAll( Arrays.asList( types ) );
64         return exceptions;
65     }
66 
67     public Method getGetter()
68     {
69         return fieldGetter;
70     }
71 
72     public void setGetter( Method getter )
73     {
74         fieldGetter = getter;
75     }
76 
77 }
