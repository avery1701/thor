66     public static class MockMethodInterceptor implements MethodInterceptor, Serializable
67     {
68 
69         private static final long serialVersionUID = -9054190871232972342L;
70 
71         private final InvocationHandler handler;
72 
73 
74         public MockMethodInterceptor( final InvocationHandler handler )
75         {
76             this.handler = handler;
77         }
78 
79         public Object intercept( final Object obj, final Method method, final Object[] args,
80                 final MethodProxy proxy ) throws Throwable
81         {
82 
83             // Bridges should be called so they can forward to the real
84             // method
85             if ( method.isBridge() )
86             {
87                 final Method m = BridgeMethodResolver.findBridgedMethod( method );
88                 return handler.invoke( obj, m, args );
89             }
90 
91             // We conveniently mock abstract methods be default
92             if ( Modifier.isAbstract( method.getModifiers() ) )
93             {
94                 return handler.invoke( obj, method, args );
95             }
96 
97             return handler.invoke( obj, method, args );
98         }
99 
100    }

103    private static final NamingPolicy ALLOWS_MOCKING_CLASSES_IN_SIGNED_PACKAGES = new DefaultNamingPolicy()
104    {
105        @Override
106        public String getClassName( final String prefix, final String source, final Object key,
107                final Predicate names )
108        {
109            return "codegen." + super.getClassName( prefix, source, key, names );
110        }
111    };

115    @SuppressWarnings( "unchecked" )
116    public T createProxy( final Class<T> toMock, final InvocationHandler handler )
117    {
118
119        final Enhancer enhancer = createEnhancer( toMock );
120
121        final MethodInterceptor interceptor = new MockMethodInterceptor( handler );
122        enhancer.setCallbackType( interceptor.getClass() );
123
124        Class mockClass;
125        try
126        {
127            mockClass = enhancer.createClass();
128        }
129        catch ( final CodeGenerationException e )
130        {
131            // ///CLOVER:OFF (don't know how to test it automatically)
132            // Probably caused by a NoClassDefFoundError, let's try EasyMock class loader
133            // instead of the default one (which is the class to mock one
134            // This is required by Eclipse Plug-ins, the mock class loader doesn't see
135            // cglib most of the time. Using EasyMock class loader solves this
136            // See issue ID: 2994002
137            enhancer.setClassLoader( getClass().getClassLoader() );
138            mockClass = enhancer.createClass();
139            // ///CLOVER:ON
140        }
141
142        Enhancer.registerCallbacks( mockClass, new Callback[] { interceptor } );
143
144        if ( ClassExtensionHelper.getCurrentConstructorArgs() != null )
145        {
146            // Really instantiate the class
147            final ConstructorArgs args = ClassExtensionHelper.getCurrentConstructorArgs();
148            Constructor cstr;
149            try
150            {
151                // Get the constructor with the same params
152                cstr = mockClass.getDeclaredConstructor( args.getConstructor().getParameterTypes() );
153            }
154            catch ( final NoSuchMethodException e )
155            {
156                // Shouldn't happen, constructor is checked when ConstructorArgs is instantiated
157                // ///CLOVER:OFF
158                throw new RuntimeException( "Fail to find constructor for param types", e );
159                // ///CLOVER:ON
160            }
161            T mock;
162            try
163            {
164                cstr.setAccessible( true ); // So we can call a protected
165                // constructor
166                mock = (T) cstr.newInstance( args.getInitArgs() );
167            }
168            catch ( final InstantiationException e )
169            {
170                // ///CLOVER:OFF
171                throw new RuntimeException( "Failed to instantiate mock calling constructor", e );
172                // ///CLOVER:ON
173            }
174            catch ( final IllegalAccessException e )
175            {
176                // ///CLOVER:OFF
177                throw new RuntimeException( "Failed to instantiate mock calling constructor", e );
178                // ///CLOVER:ON
179            }
180            catch ( final InvocationTargetException e )
181            {
182                throw new RuntimeException(
183                        "Failed to instantiate mock calling constructor: Exception in constructor",
184                        e.getTargetException() );
185            }
186            return mock;
187        }
188        else
189        {
190            // Do not call any constructor
191
192            Factory mock;
193            try
194            {
195                mock = (Factory) ClassInstantiatorFactory.getInstantiator().newInstance( mockClass );
196            }
197            catch ( final InstantiationException e )
198            {
199                // ///CLOVER:OFF
200                throw new RuntimeException( "Fail to instantiate mock for " + toMock + " on "
201                        + ClassInstantiatorFactory.getJVM() + " JVM" );
202                // ///CLOVER:ON
203            }
204
205            // This call is required. CGlib has some "magic code" making sure a
206            // callback is used by only one instance of a given class. So only
207            // the
208            // instance created right after registering the callback will get
209            // it.
210            // However, this is done in the constructor which I'm bypassing to
211            // allow class instantiation without calling a constructor.
212            // Fortunately, the "magic code" is also called in getCallback which
213            // is
214            // why I'm calling it here mock.getCallback(0);
215            mock.getCallback( 0 );
216
217            return (T) mock;
218        }
219    }
220
221    private Enhancer createEnhancer( final Class<T> toMock )
222    {
223        // Create the mock
224        final Enhancer enhancer = new Enhancer()
225        {
226            /**
227             * Filter all private constructors but do not check that there are
228             * some left
229             */
230            @SuppressWarnings( "unchecked" )
231            @Override
232            protected void filterConstructors( final Class sc, final List constructors )
233            {
234                CollectionUtils.filter( constructors, new VisibilityPredicate( sc, true ) );
235            }
236        };
237        enhancer.setSuperclass( toMock );
238
239        // ///CLOVER:OFF (I don't know how to test it automatically yet)
240        // See issue ID: 2994002
241        if ( toMock.getSigners() != null )
242        {
243            enhancer.setNamingPolicy( ALLOWS_MOCKING_CLASSES_IN_SIGNED_PACKAGES );
244        }
245        // ///CLOVER:ON
246
247        return enhancer;
248    }
