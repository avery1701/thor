16 package com.advancedpwr.record.mock;
17 
18 import java.lang.reflect.InvocationHandler;
19 import java.lang.reflect.InvocationTargetException;
20 import java.lang.reflect.Method;
21 
22 
23 public class MockRecordingInvocationHandler implements InvocationHandler
24 {
25     
26     protected BehaviorInstanceTree fieldTree;
27     protected MockBehaviorRecorder fieldRecorder;
28     
29     public MockRecordingInvocationHandler( MockBehaviorRecorder inRecorder, BehaviorInstanceTree inTree )
30     {
31         fieldRecorder = inRecorder;
32         fieldTree = inTree;
33     }
34 
35     protected void debug( Object inObject )
36     {
37         System.out.println( inObject );
38     }
39     
40     public Object invoke( Object proxy, Method method, Object[] args ) throws Throwable
41     {
42         debug( "Invoking method " + getSource().getClass().getSimpleName() + "." + method.getName() + "()" );
43         getTree().addDeclaredExceptions( method.getExceptionTypes() );
44         Object[] instrumentedArgs = instrumentArguments( method, args );
45         Object result;
46         try
47         {
48             method.setAccessible( true );
49             debug( "Proxy: " + proxy.getClass() + ", Source: " + getSource().getClass() );
50             result = method.invoke( getSource(), instrumentedArgs );
51         }
52         catch ( InvocationTargetException ite )
53         {
54             getTree().setCurrentMethod( method );
55             getTree().addAccessPaths( instrumentedArgs, ite.getTargetException() );
56             throw ite.getTargetException();
57         }
58         if ( isObjectMethod( method ) )
59         {
60             return result;
61         }
62         getTree().setCurrentMethod( method );
63         getTree().addAccessPaths( instrumentedArgs, result );
64         return instrument( result );
65     }
66 
67     protected Object[] instrumentArguments( Method method, Object[] args )
68     {
69         Object[] instrumentedArgs;
70         if ( isObjectMethod( method ) )
71         {
72             instrumentedArgs = args;
73         }
74         else
75         {
76             instrumentedArgs = instrumentedArray( args );
77         }
78         return instrumentedArgs;
79     }
80 
81     protected <T> T[] instrumentedArray( T[] inArray )
82     {
83         return getRecorder().instrumentedArray( inArray );
84     }
85 
86     protected Object getSource()
87     {
88         return getTree().getObject();
89     }
90     
91     protected BehaviorInstanceTree getTree()
92     {
93         return fieldTree;
94     }
95     
96     protected boolean isObjectMethod( Method method )
97     {
98         return "hashCode".equals( method.getName() ) || "equals".equals( method.getName() )  || "toString".equals( method.getName() ) || "finalize".equals( method.getName() );
99     }
100
101    protected MockBehaviorRecorder getRecorder()
102    {
103        return fieldRecorder;
104    }
105    
106    protected <T> T instrument( final T inObject )
107    {
108        return getRecorder().instrument( inObject );
109    }
110
111}
