16 package com.advancedpwr.record.mock;
17 
18 import java.lang.reflect.InvocationHandler;
19 import java.lang.reflect.Proxy;
20 
21 import org.easymock.internal.IProxyFactory;
22 
23 public class InterfaceProxyFactory<T> implements IProxyFactory<T>
24 {
25     protected Class fieldInterface;
26     
27     public InterfaceProxyFactory( Class inInterface )
28     {
29         fieldInterface = inInterface;
30     }
31 
32     protected Class getInterface()
33     {
34         return fieldInterface;
35     }
36 
37     @SuppressWarnings( "unchecked" )
38     public T createProxy( final Class<T> inClass, final InvocationHandler handler )
39     {
40         return (T)Proxy.newProxyInstance( inClass.getClassLoader(), new Class[]{ getInterface() }, handler );
41     }
42     
43 }
