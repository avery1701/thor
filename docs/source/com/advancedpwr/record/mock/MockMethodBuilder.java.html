16 package com.advancedpwr.record.mock;
17 
18 import java.util.Iterator;
19 
20 import com.advancedpwr.record.AccessPath;
21 import com.advancedpwr.record.methods.BuildMethodWriter;
22 
23 public class MockMethodBuilder extends BuildMethodWriter
24 {
25 
26     protected void writeInstance()
27     {
28         writeIfNotNullReturnInstance();
29         writeLine( instanceName() + " = createStrictMock( " + instanceType() + ".class )");
30     }
31     
32     protected void writePopulators()
33     {
34         for ( AccessPath result : getInstanceTree().getAccessPaths() )
35         {
36             if ( result instanceof ArgumentPath )
37             {
38                 continue;
39             }
40             BuildMethodWriter builder = createMethodBuilder( result );
41             String methodCall = instanceName() + "." + methodName( result ) + "(" + buildArguments( result ) +")";
42             if ( isVoidReturnType( result ) )
43             {
44                 if ( isException( result ) )
45                 {
46                     writeLine( methodCall );
47                     writeLine( "expectLastCall().andThrow( "+ builder.resultBuilder() + " )" );
48                 }
49                 else
50                 {
51                     writeLine(  methodCall );
52                 }
53             }
54             else
55             {
56                 if ( isException( result ) )
57                 {
58                     writeLine( "expect( " + methodCall + " ).andThrow( " + builder.resultBuilder() + " )" );
59                 }
60                 else
61                 {
62                     writeLine( "expect( " + methodCall + " ).andReturn( " + builder.resultBuilder() + " )" );
63                 }
64             }
65         }
66         
67         writeLine( "replay( " + instanceName() + " )");
68     }
69 
70     protected String methodName( AccessPath result )
71     {
72         return result.pathName();
73     }
74 
75     protected boolean isException( AccessPath result )
76     {
77         return Throwable.class.isAssignableFrom( result.getResultClass() );
78     }
79 
80     protected boolean isVoidReturnType( AccessPath result )
81     {
82         return result.getReturnType().equals( void.class ); 
83     }
84 
85     protected String buildArguments( AccessPath result )
86     {
87         StringBuffer sb = new StringBuffer( " " );
88         if ( result instanceof MethodCallPath )
89         {
90             MethodCallPath callPath = (MethodCallPath) result;
91             if( callPath.getArguments().size() == 0 )
92             {
93                 return "";
94             }
95             for ( Iterator iterator = callPath.getArguments().iterator(); iterator.hasNext(); )
96             {
97                 AccessPath path = (AccessPath) iterator.next();
98                 BuildMethodWriter builder = createMethodBuilder( path );
99                 sb.append( builder.resultBuilder() );
100                if ( iterator.hasNext() )
101                {
102                    sb.append( "," );
103                }
104                else
105                {
106                    sb.append( " " );
107                }
108            }
109        }
110        return sb.toString();
111    }
112
113}
